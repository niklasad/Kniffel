6   package kniffel.gui;
7   
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import javax.swing.JButton;
11  import javax.swing.JLabel;
12  import javax.swing.JTextField;
13  import kniffel.logiikka.Pelialusta;
14  
15  /**
16   * Kuuntelija aloitusikkunalle, joka lisää pelaajat pelialustalle.
17   *
18   * @author niklasad
19   */
20  public class AloitusIkkunanKuuntelija implements ActionListener {
21  
22      private Pelialusta alusta;
23      private JTextField pelaaja;
24      private JButton aloitusnappi;
25      private JLabel ohje;
26      private JButton lisaysnappi;
27  
28      /**
29       * konsturuktori joka saa parametreina tarvittavat komponentit
30       * muokattavaksi.
31       *
32       * @param alusta = pelialusta
33       * @param nimi = tekstikenttä johon nimi kirjoitetaan
34       * @param o = päivittyvä ohjeteksti
35       * @param a = nappi josta peli aloitetaan
36       * @param p = nappi jolla pelaaja lisätään peliin (nimi lisätään)
37       */
38      public AloitusIkkunanKuuntelija(Pelialusta alusta, JTextField nimi, JLabel o, JButton a, JButton p) {
39          this.alusta = alusta;
40          this.pelaaja = nimi;
41          this.ohje = o;
42          this.aloitusnappi = a;
43          this.lisaysnappi = p;
44      }
45  
46      @Override
47      public void actionPerformed(ActionEvent e) {
48          if (pelaaja.getText().isEmpty()) {
49              ohje.setText("Tyhjä nimi ei kelpaa.");
50          } else if (pelaaja.getText().length() > 10) {
51              ohje.setText("Nimen maksimipituus 10 merkkiä, syötä uusi");
52              pelaaja.setText("");
53          } else {
54              lisaaPelaaja();
55              pelaaja.setText("");
56              aloitusnappi.setEnabled(true);
57          }
58      }
59  
60      /**
61       * Jos pelissä ei ole vielä 4 pelaajaa, metodi lisää pelialustan listalle
62       * uuden pelaajan.
63       */
64      public void lisaaPelaaja() {
65          this.alusta.lisaaPelaaja(pelaaja.getText());
66          if (alusta.getPelaajat().size() < 4) {
67              ohje.setText("Pelaaja lisätty! Syötä seuraava nimi");
68          } else {
69              lisaysnappi.setEnabled(false);
70              ohje.setText("4 pelaajaa pelissä, aloita peli!");
71          }
72      }
73  
74  }
