6   package kniffel.gui;
7   
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import java.util.List;
11  import javax.swing.JButton;
12  import javax.swing.JFrame;
13  import javax.swing.JLabel;
14  import kniffel.logiikka.Pelialusta;
15  
16  /**
17   * Kuuntelija nopan heittämiselle.
18   *
19   * @author niklasad
20   */
21  public class HeitonKuuntelija implements ActionListener {
22  
23      private Pelialusta alusta;
24      private List<JButton> nopat;
25      private int heitto;
26      private JButton heittoNappi;
27      private JLabel status;
28      private List<JButton> napit;
29      private JFrame ikkuna;
30  
31      /**
32       * konstruktori joka saa parametreina tarvittavat komponentit.
33       *
34       * @param a = pelialusta
35       * @param nopat = noppanapit
36       * @param b = heittonappula
37       * @param s = statuslabel (päivittyvä)
38       * @param n = tulosnapit jotka avataan käyttöön tässä kuuntelijassa
39       * @param peliIkkuna = peli ikkuna, jotta se voidaan lopussa sulkea
40       * noppanapin avulla
41       */
42      public HeitonKuuntelija(Pelialusta a, List<JButton> nopat, JButton b, JLabel s, List<JButton> n, JFrame peliIkkuna) {
43          this.alusta = a;
44          this.nopat = nopat;
45          this.heitto = 0;
46          this.heittoNappi = b;
47          this.status = s;
48          this.napit = n;
49          this.ikkuna = peliIkkuna;
50      }
51  
52      /**
53       * Jos Pelialustan getKierros palauttaa 15, on peli päättynyt ja noppanapin
54       * teksti on jo muuttunut "Lopeta peli" muotoon ja tällöin sen painaminen
55       * lopettaa nykyisen pelin ja avaa uuden AloitusIkkunan.
56       *
57       * @param e = näppäintä klikattu
58       */
59      @Override
60      public void actionPerformed(ActionEvent e) {
61          if (alusta.getKierros() == 15) {
62              ikkuna.dispose();
63              new AloitusIkkuna().setVisible(true);
64          } else {
65              perusKierros();
66          }
67  
68      }
69  
70      /**
71       * Normaalit kierrostoimenpiteet Noppapainikkeelle, eli pelaajan ensimmäisen
72       * heiton jälkeen vapauttaa nopat valittaviksi ja heittää nopat ja päivittää
73       * numerot oikeiksi käyttöliittymään. 3 heiton jälkeen asettaa heittonapin
74       * pois käytöstä ja vapauttaa tulosnapit jotta tuloksen voi asettaa, sekä
75       * päivittää statuslabelin ohjeet ja nollaa vielä pelaajakohtaisen
76       * heittovuoron.
77       */
78      private void perusKierros() {
79          if (heitto < 2) {
80              if (heitto == 0) {
81                  vapautaNopat();
82              }
83              this.heitaJaPaivitaNopat();
84          } else {
85              this.heitaJaPaivitaNopat();
86              this.heittoNappi.setEnabled(false);
87              vapautaTulosNapit();
88              this.status.setText("Aseta pisteet");
89              this.heitto = 0;
90          }
91      }
92  
93      /**
94       * Heittää nopat Pelialustalla ja päivittää arvot käyttöliittymään
95       */
96      private void heitaJaPaivitaNopat() {
97          this.alusta.heitaNopat();
98          heitto++;
99          paivitaNopat();
100         paivitaStatus();
101     }
102 
103     private void vapautaTulosNapit() {
104         for (JButton nappi : napit) {
105             nappi.setEnabled(true);
106         }
107     }
108 
109     private void vapautaNopat() {
110         for (JButton noppa : nopat) {
111             noppa.setEnabled(true);
112         }
113     }
114 
115     private void paivitaNopat() {
116         int x = 0;
117         for (JButton noppa : nopat) {
118             noppa.setText(alusta.getNopat().get(x).toString());
119             x++;
120         }
121     }
122 
123     private void paivitaStatus() {
124         if (this.heitto == 1) {
125             this.status.setText("2 heittoa jäljellä");
126         } else if (this.heitto == 2) {
127             this.status.setText("1 heitto jäljellä");
128         }
129     }
130 
131 }
